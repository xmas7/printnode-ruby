def to_hash
  hash = {}
  hash['printerId'] = @printer_id
  hash['title'] = @title
  hash['contentType'] = @content_type
  if @content_type.match('base64$')
    hash ['content'] = base64.encode(IO.read(@content))
  else
    hash ['content'] = @content
  end
  hash['source'] = @source
  hash
end
